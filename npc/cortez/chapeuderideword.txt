	mes "Eu tenho alguns itens interessantes aqui, deseja dar uma olhada?";

	{
		mes "[Viajante]";
		mes "Tudo bem, estarei aqui se você mudar de idéia.";

	next;
	switch(select("Chapéu de Rideword:Coroa De Áries:Nenhum"))
	{
	case 1:
		mes "[Chapéu de Rideword]";
		mes "Para poder fazer este chapéu, eu vou precisar de:";
		mes "500x "+getitemname(1097);
		mes "130x "+getitemname(7015);
		mes "10x "+getitemname(4185);
		mes "100x "+getitemname(7449);
		mes "10x "+getitemname(4388);
		mes "15x "+getitemname(2221);
		mes "20000 Zenys";
		next;

		mes "Deseja fazer esta quest?";
		next;
		if( select("Sim!:Agora não...") == 2 )

			mes "[Viajante]";
			mes "Tudo bem... Até mais!";
			close;
		}
		if(countitem(1097) >= 500 && countitem(7015) >= 130 && countitem(4185) >= 10 && countitem(7449) >= 100 && countitem(4388) >= 10 && countitem(2221) >= 15 && Zeny >= 20000)
		{
			delitem 1097,500;
			delitem 7015,130;
			delitem 4185,10;
			delitem 7449,100;
			delitem 4388,10;
			delitem 2221,15;
			set Zeny,Zeny-20000;
			getitem 5208,1;
			mes "[Viajante]";
			mes "Parabéns! Aqui está!";
			close;
		}
		break;
	case 2:
		mes "[Coroa de Áries]";
		mes "Para poder fazer este chapéu, eu vou precisar de:";
		mes "5x "+getitemname(4109);
		mes "5x "+getitemname(4099);
		mes "1x "+getitemname(1258);
		mes "30x "+getitemname(969);
		mes "300x "+getitemname(7106);
		mes "500x "+getitemname(947);
		mes "20000 Zenys";
		next;
		mes "[Viajante]";
		mes "Deseja fazer esta quest?";
		next;
		if( select("Sim!:Agora não...") == 2 )
		{
			mes "[Viajante]";
			mes "Tudo bem... Até mais!";
			close;
		}
		if(countitem(4109) >= 10 && countitem(4099) >= 10 && countitem(1258) >= 10 && countitem(969) >= 30 && countitem(7106) >= 300 && countitem(947) >= 500 && Zeny >= 20000)

			delitem 4109,10;

			delitem 1258,10;
			delitem 969,30;
			delitem 7106,300;
			delitem 947,500;
			set Zeny,Zeny-20000;
			getitem 5546,1;
			mes "[Viajante]";
			mes "Parabéns! Aqui está!";
			close;
		}
		break;
	case 3:
		mes "[Viajante]";
		mes "Tudo bem, estarei aqui se você mudar de idéia.";
		close;
	}
	mes "[Viajante]";
	mes "Me desculpe, mas você não tem todos os itens necessários para esta quest.";

}
