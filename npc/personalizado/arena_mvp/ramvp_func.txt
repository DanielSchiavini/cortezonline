function	script	Ramvp	{

	switch(getarg(0))
	{

	mes "[MvP Alpha]";
		mes "Até mais!";
			return;
case 2:

	mes "[MvP Beta]";
		mes "Até mais!";
			return;
case 3:
	mes "[MvP Theta]";
		mes "Até mais!";
			return;
case 4:
	mes "[MvP Epsilon]";
		mes "Até mais!";
			return;
	}
}


function	script	Ramvp_M	{

	if(!getarg(0,0))
	{
		mes "Do que você precisa?.";
		switch(select("- Monstros Díficeis:- Curar-me:- Sair"))
		{
	case 1:
		return;

	case 2:
		percentheal 100,100;
		close;

	case 3:
		warp "prontera",156,179;
		close;
		}
	}

	mes "Do que você precisa?.";
	if(select("- Cura:- Sair")==1)
	{
		percentheal 100,100;
		close;
	}
		warp "prontera",156,179;
		close;

}

function	script	Ramvp_Main	{

	mes "[Guia do MvP]";

function	script	MenuLvl	{

	if (getmapusers(getarg(0)) > getarg(1))
	{
		mes "[MvP "+getarg(3)+"]";
		mes "Desculpe mas está arena está atualmente cheia, porém você pode escolher outra!";
		close;
	} callfunc("Ramvp",getarg(2));
	warp getarg(0),102,102;
	close;
}
