function Show_time {

		for(set .@i,0; .@i <= getarraysize($week_days$); set .@i,.@i+1)
		{
			if($agit_hora_start[.@i] != -1)
				set .@horarios$[.@i],$week_days$[.@i];
			else
				set .@horarios$[.@i],"";
		}

		for(set @i,0;@i <= getarraysize($agit_hora_start);set @i,@i+1)
		{
			if(.@horarios$[@i] != "")
			{
				if($agit_hora_start[@i] < 10)
					set @agit_hora_start$,"0";
				else
					set @agit_hora_start$,"";

				if($agit_hora_end[@i] < 10)
					set @agit_hora_end$,"0";
				else
					set @agit_hora_end$,"";

				if($agit_min_start[@i] < 10)
					set @agit_min_start$,"0";
				else
					set @agit_min_start$,"";

				if($agit_min_end[@i] < 10)
					set @agit_min_end$,"0";
				else
					set @agit_min_end$,"";

				mes .@horarios$[@i]+" - "+$agit_hora_start[@i]+@agit_hora_start$+":"+$agit_min_start[@i]+@agit_min_start$+" às "+$agit_hora_end[@i]+@agit_hora_end$+":"+$agit_min_end[@i]+@agit_min_end$;
			}
		}
	return;
